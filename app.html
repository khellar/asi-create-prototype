# research-scout.flow — generated by Research scout
from asi import schedule, tools, llm

@schedule("0 7 * * *")            # daily 7:00am
def research_scout():
    hits = tools.web_search("new web3 projects this week", provider="perplexity")
    hits = [h for h in hits if h.volume_24h_usd >= 1_000_000]   # ← your filter
    memo = llm.summarize(hits, prompt="3-bullet memo per project, focus on traction")
    tools.deliver(memo, to="email", subject="Daily research")
